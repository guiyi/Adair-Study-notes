### set中的值是不能重复的！

```shell
192.168.0.182:6379[3]> sadd hello world # set集合中添加元素
(integer) 1
192.168.0.182:6379[3]> sadd hello world1
(integer) 1
192.168.0.182:6379[3]> smembers hello # 查看指定set的所有值
1) "world1"
2) "world"
192.168.0.182:6379[3]> sismember hello world2  #判断某一个值是否存在
(integer) 0
192.168.0.182:6379[3]> sismember hello world
(integer) 1

#####################################################################
#rem

192.168.0.182:6379[3]> sadd myset h1
(integer) 1
192.168.0.182:6379[3]> sadd myset h2
(integer) 1
192.168.0.182:6379[3]> sadd myset h3
(integer) 1
192.168.0.182:6379[3]> sadd myset h1
(integer) 0
192.168.0.182:6379[3]> smembers myset
1) "h1"
2) "h3"
3) "h2"
192.168.0.182:6379[3]> srem myset h2
(integer) 1
192.168.0.182:6379[3]> smembers myset
1) "h3"
2) "h1"
192.168.0.182:6379[3]> scard myset
(integer) 2

#####################################################################
#srandmember 无需集合，抽随机

192.168.0.182:6379[3]> srandmember myset
"h3"
192.168.0.182:6379[3]> srandmember myset
"h1"
192.168.0.182:6379[3]> smembers myset
1) "h3"
2) "h2"
3) "h4"
4) "h1"
5) "h5"

#####################################################################
#spop 删除key,随机删除key！
192.168.0.182:6379[3]> spop myset
"h5"
192.168.0.182:6379[3]> spop myset
"h1"
192.168.0.182:6379[3]> smembers myset
1) "h4"
2) "h2"
3) "h3"

#####################################################################
# smove 将一个指定的值，移动到另外一个set集合

192.168.0.182:6379[3]> smove myset myset1 "h2"
(integer) 1
192.168.0.182:6379[3]> smembers myset
1) "h4"
2) "h3"
192.168.0.182:6379[3]> smembers myset1
1) "h2"

#####################################################################
# 共同关注（并集）

192.168.0.182:6379[3]> sadd key1 a
(integer) 1
192.168.0.182:6379[3]> sadd key1 b
(integer) 1
192.168.0.182:6379[3]> sadd key1 c
(integer) 1
192.168.0.182:6379[3]> sadd key2 c
(integer) 1
192.168.0.182:6379[3]> sadd key2 d
(integer) 1
192.168.0.182:6379[3]> sadd key2 e
(integer) 1

192.168.0.182:6379[3]> sdiff key1 key2  #差集
1) "a"
2) "b"
192.168.0.182:6379[3]> sdiff key2 key1
1) "d"
2) "e"
192.168.0.182:6379[3]> sinter key1 key2 #交集 。 共同关注
1) "c"
192.168.0.182:6379[3]> sunion key1 key2 #并集
1) "c"
2) "b"
3) "d"
4) "a"
5) "e"
```
